__Rodar o MySQL
__Para Wordpress
docker run \
    --network mysql \
    -e MYSQL_ROOT_PASSWORD=my-password \
    -e MYSQL_DATABASE=wordpress \
    -e MYSQL_USER=root \
    -e MYSQL_PASSWORD=my-password \
    --name mysql.5.7 \
    -d mysql:5.7

__Mais simples
docker run \

    -d mysql:5.7

__Rodar o PHPmyadmin
docker run \
    --network mysql \
    -p 8080:80 \
    -e PMA_HOST=mysql.5.7 \
